\newcommand{\pic}[3]{
	\vspace{2.5mm}
	\begin{center}
		\tcbox[sharp corners, boxrule=0.5mm, boxsep=0mm, colframe=black, colback=white, width=0.7\textwidth]{\includegraphics[width=0.85\textwidth]{#1}}
		\captionof{figure}{#2}
		\label{#3}
	\end{center}
	\vspace{2.5mm}
	}

\newcommand{\subpic}[3]{
	\begin{subfigure}[h]{0.5\textwidth}
		\centering

		\caption{#2}
		\label{#3}
	\end{subfigure}
	}
